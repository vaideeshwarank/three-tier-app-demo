5) npm install

7) from frontend directory:
docker build -t my-react-app -f Frontend_Dockerfile.dockerfile .
8) run 
docker run -p 3000:3000 my-react-app
9) from backend directory:
docker build -t my-backend-app -f Backend_Dockerfile .
8) run
